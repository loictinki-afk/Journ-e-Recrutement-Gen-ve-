      // On récupère le bouton correspondant

      var btn = document.querySelector('[data-session="' + sessionId + '"]');

      if (btn && btn.dataset.locked === "true") {

        alert("Ce créneau sera ouvert dès que le créneau précédent sera complet. Merci de choisir un créneau déjà ouvert.");

        return;

      };
